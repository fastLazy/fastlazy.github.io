useLazyQuery

import { gql } from "@apollo/client";

export const GET_USER_LIST = gql`
  input UserInput {
    name: String!
  }

  query GetList($where: UserInput!, $limit: Int!, $offset: Int!) {
    users_aggregate(where: $where, limit: $limit, offset: $offset) {
      nodes {
        email
        id
        name
        password
      }
      aggregate {
        count
      }
    }
  }
`;
import { useLazyQuery } from "@apollo/client";

const [getUsers, { loading }] = useLazyQuery(GET_USER_LIST, {
  fetchPolicy: "network-only",
});
const getList = async (params: {
  current: number,
  pageSize: number,
  name: string,
}) => {
  const result = await getUsers({
    variables: {
      offset: (params.current - 1) * params?.pageSize, // 分页偏移量:(当前页码 - 1) * pageSize
      limit: params?.pageSize,
      where: {
        name: {
          _like: `%${params.name ?? ""}%`,
        },
      },
    },
  });

  return {
    data: result.data?.users_aggregate?.nodes,
    total: result.data?.users_aggregate?.aggregate?.count,
    success: true,
  };
};
